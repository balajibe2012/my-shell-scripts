	3) Analysis query 
	Get top 5 brands with most models
	SELECT Make_Name, COUNT(DISTINCT Model_ID) AS model_count
	FROM vehicle_models
	GROUP BY Make_Name
	ORDER BY model_count DESC
	LIMIT 5;
